快速开始

1. 引入CSS模块

<link rel="stylesheet" href="css/modules/variables.css"> <link rel="stylesheet" href="css/modules/base.css"> <link rel="stylesheet" href="css/modules/layout.css"> <link rel="stylesheet" href="css/modules/forms.css"> <link rel="stylesheet" href="css/modules/navigation.css"> <link rel="stylesheet" href="css/modules/display.css"> <link rel="stylesheet" href="css/modules/feedback.css"> <link rel="stylesheet" href="css/modules/tabs-browser.css">

2. 引入JS模块

<script src="js/modules/theme.js"></script> <script src="js/modules/tabs.js"></script> <script src="js/modules/modal.js"></script> <script src="js/modules/feedback.js"></script>

3. 使用组件

<!-- 按钮 --> <button class="btn-main">主按钮</button> <!-- 卡片 --> <div class="card"> <div class="card-header">标题</div> <p>内容</p> </div> <!-- 主题切换 --> <button onclick="setTheme('dark')">深色模式</button>
